## 一、测试策略概述

### 1.1 测试层级

```
┌─────────────────────────────────────────┐
│              端到端测试 (E2E)            │
│         模拟用户操作，验证完整流程        │
├─────────────────────────────────────────┤
│              集成测试                    │
│         验证模块间交互和数据流            │
├─────────────────────────────────────────┤
│              单元测试                    │
│         验证单个函数/组件的正确性         │
├─────────────────────────────────────────┤
│              静态检查                    │
│         代码规范、类型检查、安全扫描      │
└─────────────────────────────────────────┘
```

### 1.2 测试原则

- **自动化优先**：核心功能必须有自动化测试覆盖
- **分层测试**：不同层级关注不同问题
- **持续集成**：每次提交都运行测试
- **测试即文档**：测试用例描述功能行为

## 二、单元测试规范

### 2.1 测试框架

- **小程序端**：使用 Jest + miniprogram-simulate
- **云函数端**：使用 Jest

### 2.2 测试文件命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 测试文件 | `{被测文件}.test.js` | `userService.test.js` |
| 测试目录 | `__tests__/` | `services/__tests__/` |

### 2.3 测试文件位置

```
services/
├── userService.js
├── userService.test.js       # 同目录测试文件
└── __tests__/                # 或测试目录
    └── userService.test.js

utils/
├── validator.js
└── validator.test.js
```

### 2.4 测试用例结构

```javascript
// userService.test.js

const UserService = require('../userService')

describe('UserService', () => {
  let userService

  // 测试前准备
  beforeEach(() => {
    userService = UserService.getInstance()
  })

  // 测试后清理
  afterEach(() => {
    jest.clearAllMocks()
  })

  describe('getUserInfo', () => {
    test('应该返回用户信息', async () => {
      // 准备
      const mockUser = { _id: '123', nickName: '张三' }
      jest.spyOn(userService, 'getUserInfo').mockResolvedValue(mockUser)

      // 执行
      const result = await userService.getUserInfo()

      // 断言
      expect(result).toEqual(mockUser)
      expect(result.nickName).toBe('张三')
    })

    test('用户不存在时应该抛出错误', async () => {
      // 准备
      jest.spyOn(userService, 'getUserInfo').mockRejectedValue(new Error('用户不存在'))

      // 执行 & 断言
      await expect(userService.getUserInfo()).rejects.toThrow('用户不存在')
    })
  })

  describe('updateUserInfo', () => {
    test('应该成功更新用户信息', async () => {
      // 准备
      const updates = { nickName: '李四' }
      const mockResult = { _id: '123', nickName: '李四' }
      jest.spyOn(userService, 'updateUserInfo').mockResolvedValue(mockResult)

      // 执行
      const result = await userService.updateUserInfo(updates)

      // 断言
      expect(result.nickName).toBe('李四')
    })
  })
})
```

### 2.5 测试用例命名规范

- 使用 `describe` 分组组织测试
- 使用 `test` 或 `it` 定义测试用例
- 测试描述使用**中文**，清晰表达测试意图
- 遵循 **"应该...当..."** 或 **"应该..."** 格式

```javascript
// 好的测试命名
describe('BookingService', () => {
  describe('createBooking', () => {
    test('应该成功创建预约', async () => { })
    test('应该抛出错误当时间冲突时', async () => { })
    test('应该抛出错误当信誉分不足时', async () => { })
  })
})

// 不好的测试命名
describe('BookingService', () => {
  test('test1', () => { })
  test('createBooking', () => { })
  test('error', () => { })
})
```

### 2.6 测试覆盖要求

| 模块类型 | 覆盖率要求 | 说明 |
|---------|-----------|------|
| 核心服务类 | ≥ 80% | userService, bookingService 等 |
| 工具函数 | ≥ 70% | validator, dateUtil 等 |
| 云函数处理器 | ≥ 60% | 核心业务逻辑 |
| 页面逻辑 | ≥ 50% | 关键业务页面 |

// meetingroomService.test.js

const cloud = require('wx-server-sdk')
const meetingroomService = require('./index')

describe('meetingroomService', () => {
  describe('meetingroom_booking_create', () => {
    test('应该成功创建预约', async () => {
      // 准备
      const event = {
        action: 'meetingroom_booking_create',
        params: {
          roomId: 'room_001',
          date: '2026-04-22',
          startTime: '09:00',
          endTime: '11:00',
          purpose: '项目讨论'
        }
      }

      // 执行
      const result = await meetingroomService.main(event, {})

      // 断言
      expect(result.code).toBe(200)
      expect(result.data.bookingId).toBeDefined()
    })

    test('应该返回错误当时间冲突时', async () => {
      // 准备
      const event = {
        action: 'meetingroom_booking_create',
        params: {
          roomId: 'room_001',
          date: '2026-04-22',
          startTime: '09:00',
          endTime: '11:00',
          purpose: '项目讨论'
        }
      }

      // 模拟时间冲突
      jest.spyOn(bookingService, 'checkConflict').mockResolvedValue(true)

      // 执行
      const result = await meetingroomService.main(event, {})

      // 断言
      expect(result.code).toBe(1001)
      expect(result.message).toContain('时间冲突')
    })
  })
})

## 四、端到端测试规范

### 4.1 测试工具

- 使用微信开发者工具的自动化测试能力
- 使用 Miniprogram Automator

### 4.2 测试场景

```javascript
// e2e/booking.test.js

const automator = require('miniprogram-automator')

describe('预约流程', () => {
  let miniProgram

  beforeAll(async () => {
    miniProgram = await automator.launch({
      projectPath: './miniprogram'
    })
  }, 30000)

  afterAll(async () => {
    await miniProgram.close()
  })

  test('用户应该能够完成会议室预约', async () => {
    // 1. 进入首页
    const page = await miniProgram.reLaunch('/pages/index/index')
    await page.waitFor(1000)

    // 2. 点击预约入口
    const bookingBtn = await page.$('.quick-action[data-id="booking"]')
    await bookingBtn.tap()
    await page.waitFor(1000)

    // 3. 选择会议室
    const roomCard = await page.$('.room-card')
    await roomCard.tap()
    await page.waitFor(1000)

    // 4. 选择日期和时间
    const dateCell = await page.$('.calendar__day--available')
    await dateCell.tap()
    
    const timeSlot = await page.$('.time-slot--available')
    await timeSlot.tap()

    // 5. 填写预约信息
    const purposeInput = await page.$('input[name="purpose"]')
    await purposeInput.input('项目讨论')

    // 6. 提交预约
    const submitBtn = await page.$('.btn-submit')
    await submitBtn.tap()
    await page.waitFor(2000)

    // 7. 验证预约成功
    const successPage = await miniProgram.currentPage()
    expect(successPage.path).toContain('success')
  }, 60000)
})
```

## 六、测试自动化

### 6.1 CI/CD 集成

```yaml
# .github/workflows/test.yml
name: Test

on:
  push:
    branches: [ dev, main ]
  pull_request:
    branches: [ dev, main ]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3

    - name: Setup Node.js
      uses: actions/setup-node@v3
      with:
        node-version: '18'

    - name: Install dependencies
      run: npm ci

    - name: Run lint
      run: npm run lint

    - name: Run unit tests
      run: npm run test:unit

    - name: Run integration tests
      run: npm run test:integration

    - name: Upload coverage
      uses: codecov/codecov-action@v3
```

### 6.2 测试脚本

```json
// package.json
{
  "scripts": {
    "test": "npm run test:unit && npm run test:integration",
    "test:unit": "jest --testPathPattern='.test.js$' --coverage",
    "test:integration": "jest --testPathPattern='.integration.test.js$'",
    "test:e2e": "jest --testPathPattern='.e2e.test.js$' --testTimeout=60000",
    "test:watch": "jest --watch",
    "test:coverage": "jest --coverage"
  }
}
```
