## 一、开发规范

### 1.1 代码风格规范

#### 1.1.1 缩进与格式

- **缩进**：使用 2 个空格（禁止使用 Tab）
- **换行**：Unix 风格换行符（LF）
- **行尾**：语句末尾不使用分号（JavaScript）
- **行长**：单行代码不超过 100 个字符
- **空行**：
  - 函数之间保留 1 个空行
  - 逻辑块之间保留 1 个空行
  - 文件末尾保留 1 个空行

```javascript
// 正确示例
Page({
  data: {
    userInfo: null,
    meetingRooms: []
  },

  onLoad() {
    this.initData()
  },

  initData() {
    // 初始化逻辑
    this.fetchMeetingRooms()
  }
})
```

#### 1.1.2 引号使用

- **字符串**：统一使用单引号 `'string'`
- **模板字符串**：使用反引号 `` `template` ``
- **JSON**：使用双引号（符合 JSON 标准）

```javascript
// 正确示例
const name = '红芯通'
const greeting = `欢迎使用${name}`
```

#### 1.1.3 括号与空格

```javascript
// 正确示例
if (condition) {
  doSomething()
}

function foo(a, b) {
  return a + b
}

// 对象定义
const obj = {
  key: 'value',
  num: 123
}

// 数组定义
const arr = [1, 2, 3]
```

### 1.2 命名规则

#### 1.2.1 文件命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 页面文件 | 小写 + 连字符 | `meeting-room.wxml` |
| 组件文件 | 小写 + 连字符 | `custom-button.wxml` |
| JS 文件 | 小写 + 连字符 | `user-service.js` |
| 样式文件 | 小写 + 连字符 | `common-style.wxss` |
| 图片资源 | 小写 + 连字符 + 用途 | `icon-home-active.png` |

#### 1.2.2 变量命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 普通变量 | 小驼峰 | `userName`, `meetingRoomList` |
| 常量 | 全大写 + 下划线 | `MAX_RETRY_COUNT`, `API_BASE_URL` |
| 布尔值 | 前缀 is/has/can | `isLoading`, `hasPermission`, `canSubmit` |
| 私有变量 | 下划线前缀 | `_internalData`, `_cacheMap` |
| 全局变量 | 大驼峰 | `AppConfig`, `GlobalData` |

#### 1.2.3 函数命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 获取数据 | get + 名词 | `getUserInfo()`, `getMeetingRoomList()` |
| 设置数据 | set + 名词 | `setUserInfo()`, `setCurrentRoom()` |
| 处理动作 | 动词 + 名词 | `handleSubmit()`, `processBooking()` |
| 异步操作 | 动词 + 名词 + Async | `fetchDataAsync()`, `submitFormAsync()` |
| 事件处理 | handle + 事件名 | `handleTap()`, `handleInputChange()` |
| 初始化 | init + 名词 | `initPage()`, `initUserData()` |
| 检查/判断 | is/has/can + 名词 | `isValidTime()`, `hasConflict()` |

#### 1.2.4 类/组件命名

| 类型 | 命名规范 | 示例 |
|------|----------|------|
| 组件类名 | 大驼峰 + Component 后缀 | `MeetingRoomCardComponent` |
| 服务类名 | 大驼峰 + Service 后缀 | `UserService`, `BookingService` |

#### 1.4.2 模块职责规范

| 层级 | 职责 | 禁止行为 |
|------|------|----------|
| **页面层 (pages)** | 页面渲染、用户交互、调用服务层 | 直接调用 API、直接操作数据库 |
| **组件层 (components)** | UI 渲染、事件触发、接收 props | 直接修改全局状态、直接调用服务层 |
| **服务层 (services)** | 业务逻辑、数据处理、API 调用 | 操作 DOM、处理 UI 相关逻辑 |
| **工具层 (utils)** | 通用函数、数据转换、格式处理 | 包含业务逻辑、依赖具体页面 |
| **配置层 (config)** | 全局配置、常量定义 | 包含动态数据、业务逻辑 |
| **模型层 (models)** | 数据结构定义、类型约束 | 包含业务方法、数据操作 |

#### 1.4.3 模块依赖规则

```
依赖方向（只能向下依赖，禁止循环依赖）：

pages → components
    ↓
services → utils
    ↓
config

models 可被任意层级引用
```

#### 1.4.5 组件开发规范

```javascript
// components/meeting-room-card/meeting-room-card.js

/**
 * 会议室卡片组件
 */
Component({
  // 属性定义
  properties: {
    roomInfo: {
      type: Object,
      value: {},
      observer(newVal) {
        this._processRoomData(newVal)
      }
    },
    showStatus: {
      type: Boolean,
      value: true
    }
  },

  // 内部数据
  data: {
    processedData: {}
  },

  // 生命周期
  attached() {
    this._processRoomData(this.properties.roomInfo)
  },

  // 方法定义
  methods: {
    /**
     * 处理会议室数据
     * @private
     */
    _processRoomData(roomInfo) {
      if (!roomInfo) return
      
      const processedData = {
        ...roomInfo,
        capacityText: `${roomInfo.capacity}人`,
        statusText: this._getStatusText(roomInfo.status)
      }
      
      this.setData({ processedData })
    },

    /**
     * 获取状态文本
     * @private
     */
    _getStatusText(status) {
      const statusMap = {
        available: '可预约',
        occupied: '已占用',
        maintenance: '维护中'
      }
      return statusMap[status] || '未知'
    },

    /**
     * 点击卡片事件
     */
    onCardTap() {
      this.triggerEvent('tap', { roomId: this.properties.roomInfo._id })
    },

    /**
     * 点击预约按钮
     */
    onBookTap() {
      this.triggerEvent('book', { roomId: this.properties.roomInfo._id })
    }
  }
})
```

### 1.5 错误处理机制

#### 1.5.1 错误分类

| 错误类型 | 说明 | 处理方式 |
|----------|------|----------|
| 业务错误 | 业务逻辑验证失败 | 提示用户，提供解决方案 |
| 网络错误 | 请求超时、无网络 | 自动重试或提示检查网络 |
| 系统错误 | 服务器内部错误 | 记录日志，提示稍后再试 |
| 权限错误 | 用户无权限操作 | 提示权限不足，引导申请权限 |

#### 1.5.2 统一错误处理

```javascript
// utils/errorHandler.js

/**
 * 错误码定义
 */
const ErrorCode = {
  // 业务错误 (1000-1999)
  TIME_CONFLICT: { code: 1001, message: '该时间段已被预约' },
  INVALID_PARAMS: { code: 1002, message: '参数格式不正确' },
  
  // 网络错误 (2000-2999)
  NETWORK_ERROR: { code: 2001, message: '网络连接失败，请检查网络' },
  REQUEST_TIMEOUT: { code: 2002, message: '请求超时，请稍后重试' },
  
  // 系统错误 (3000-3999)
  SERVER_ERROR: { code: 3001, message: '服务器繁忙，请稍后重试' },
  DATABASE_ERROR: { code: 3002, message: '数据操作失败' },
  
  // 权限错误 (4000-4999)
  NO_PERMISSION: { code: 4001, message: '暂无权限执行此操作' },
  NOT_LOGIN: { code: 4002, message: '请先登录' }
}

/**
 * 统一错误处理器
 */
class ErrorHandler {
  /**
   * 处理错误
   * @param {Error} error - 错误对象
   * @param {Object} options - 处理选项
   */
  static handle(error, options = {}) {
    const { showToast = true, fallback = null } = options
    
    // 记录错误日志
    this._logError(error)
    
    // 获取错误信息
    const errorInfo = this._parseError(error)
    
    // 显示提示
    if (showToast) {
      wx.showToast({
        title: errorInfo.message,
        icon: 'none',
        duration: 2000
      })
    }
    
    // 执行降级方案
    if (fallback) {
      fallback(errorInfo)
    }
    
    return errorInfo
  }

  /**
   * 解析错误
   * @private
   */
  static _parseError(error) {
    if (error.code && ErrorCode[error.code]) {
      return ErrorCode[error.code]
    }
    
    // 网络错误判断
    if (error.message && error.message.includes('network')) {
      return ErrorCode.NETWORK_ERROR
    }
    
    // 默认错误
    return {
      code: 9999,
      message: error.message || '未知错误'
    }
  }

  /**
   * 记录错误日志
   * @private
   */
  static _logError(error) {
    console.error('[Error]', error)
    // 可以在这里接入日志上报服务
  }
}

module.exports = { ErrorHandler, ErrorCode }
```

#### 1.5.3 使用示例

```javascript
// pages/booking/booking.js
const { ErrorHandler } = require('../../utils/errorHandler')
const bookingService = require('../../services/bookingService')

Page({
  async submitBooking() {
    try {
      const result = await bookingService.createBooking(this.data.formData)
      wx.showToast({ title: '预约成功', icon: 'success' })
      wx.navigateBack()
    } catch (error) {
      ErrorHandler.handle(error, {
        showToast: true,
        fallback: (errorInfo) => {
          // 特定错误的降级处理
          if (errorInfo.code === 1001) {
            this.setData({ showTimeConflictModal: true })
          }
        }
      })
    }
  }
})
```

## 五、全局用户信息管理规范

### 5.1 核心原则

**⚠️ 强制要求：涉及全局信息（如用户信息）时，所有功能模块必须统一从用户模块调取，严禁在各模块中分散存储全局信息。**

### 5.2 用户信息架构

```
┌─────────────────────────────────────────────────────────────┐
│                    全局用户信息管理架构                       │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   ┌─────────────┐     ┌─────────────┐     ┌─────────────┐  │
│   │   UserStore │◄────│  UserService │◄────│  云数据库   │  │
│   │  (全局状态)  │     │  (业务逻辑)  │     │  (users)    │  │
│   └──────┬──────┘     └─────────────┘     └─────────────┘  │
│          │                                                  │
│          │ 统一获取用户信息                                   │
│          ▼                                                  │
│   ┌─────────────┐     ┌─────────────┐     ┌─────────────┐  │
│   │  预约模块    │     │  课表模块    │     │  其他模块    │  │
│   │  Booking    │     │  Schedule   │     │   Others    │  │
│   └─────────────┘     └─────────────┘     └─────────────┘  │

│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 5.3 UserStore 实现

```javascript
// stores/userStore.js

/**
 * 用户状态管理
 * 全局唯一的用户信息存储中心
 */

const UserService = require('../services/userService')

class UserStore {
  constructor() {
    // 用户信息（受保护，禁止直接修改）
    this._userInfo = null
    
    // 登录状态
    this._isLogin = false
    
    // 权限信息
    this._permissions = []
    
    // 监听器列表
    this._listeners = []
  }

  /**
   * 获取单例实例
   */
  static getInstance() {
    if (!UserStore._instance) {
      UserStore._instance = new UserStore()
    }
    return UserStore._instance
  }

  /**
   * 获取用户信息（只读）
   * @returns {Object|null}
   */
  get userInfo() {
    return this._userInfo ? { ...this._userInfo } : null
  }

  /**
   * 获取登录状态
   * @returns {boolean}
   */
  get isLogin() {
    return this._isLogin
  }

  /**
   * 获取用户ID
   * @returns {string|null}
   */
  get userId() {
    return this._userInfo ? this._userInfo._id : null
  }

  /**
   * 获取用户角色
   * @returns {string|null}
   */
  get userRole() {
    return this._userInfo ? this._userInfo.role : null
  }

  /**
   * 获取OpenID
   * @returns {string|null}
   */
  get openid() {
    return this._userInfo ? this._userInfo.openid : null
  }

  /**
   * 初始化用户信息
   */
  async init() {
    try {
      // 从本地缓存读取
      const cachedUser = wx.getStorageSync('userInfo')
      if (cachedUser) {
        this._userInfo = cachedUser
        this._isLogin = true
        this._notifyListeners()
      }

      // 如果已登录，刷新用户信息
      if (this._isLogin) {
        await this.refreshUserInfo()
      }
    } catch (error) {
      console.error('[UserStore] 初始化失败:', error)
    }
  }

  /**
   * 设置用户信息
   * @param {Object} userInfo
   * @private
   */
  _setUserInfo(userInfo) {
    this._userInfo = userInfo
    this._isLogin = !!userInfo
    
    // 持久化到本地
    if (userInfo) {
      wx.setStorageSync('userInfo', userInfo)
    } else {
      wx.removeStorageSync('userInfo')
    }
    
    // 通知监听器
    this._notifyListeners()
  }

  /**
   * 刷新用户信息
   */
  async refreshUserInfo() {
    try {
      const userService = UserService.getInstance()
      const userInfo = await userService.getCurrentUser()
      this._setUserInfo(userInfo)
      return userInfo
    } catch (error) {
      console.error('[UserStore] 刷新用户信息失败:', error)
      throw error
    }
  }

  /**
   * 更新用户信息
   * @param {Object} updates
   */
  async updateUserInfo(updates) {
    try {
      const userService = UserService.getInstance()
      const updatedInfo = await userService.updateUser(updates)
      this._setUserInfo({ ...this._userInfo, ...updatedInfo })
      return updatedInfo
    } catch (error) {
      console.error('[UserStore] 更新用户信息失败:', error)
      throw error
    }
  }

  /**
   * 登录
   */
  async login() {
    try {
      const userService = UserService.getInstance()
      const userInfo = await userService.wxLogin()
      this._setUserInfo(userInfo)
      return userInfo
    } catch (error) {
      console.error('[UserStore] 登录失败:', error)
      throw error
    }
  }

  /**
   * 登出
   */
  logout() {
    this._setUserInfo(null)
    // 清除其他相关缓存
    wx.removeStorageSync('token')
  }

  /**
   * 检查权限
   * @param {string} permission
   * @returns {boolean}
   */
  hasPermission(permission) {
    if (!this._userInfo) return false
    if (this._userInfo.role === 'superAdmin') return true
    return this._permissions.includes(permission)
  }

  /**
   * 订阅用户信息变化
   * @param {Function} callback
   * @returns {Function} 取消订阅函数
   */
  subscribe(callback) {
    this._listeners.push(callback)
    
    // 立即执行一次回调，传递当前状态
    callback({
      userInfo: this.userInfo,
      isLogin: this.isLogin
    })
    
    // 返回取消订阅函数
    return () => {
      const index = this._listeners.indexOf(callback)
      if (index > -1) {
        this._listeners.splice(index, 1)
      }
    }
  }

  /**
   * 通知所有监听器
   * @private
   */
  _notifyListeners() {
    const state = {
      userInfo: this.userInfo,
      isLogin: this.isLogin
    }
    
    this._listeners.forEach(callback => {
      try {
        callback(state)
      } catch (error) {
        console.error('[UserStore] 监听器执行失败:', error)
      }
    })
  }
}

// 初始化单例
UserStore._instance = null

module.exports = UserStore
```

### 5.4 在页面中使用

```javascript
// pages/booking/booking.js

const UserStore = require('../../stores/userStore')
const BookingService = require('../../services/bookingService')

Page({
  data: {
    userInfo: null,
    isLogin: false,
    bookingList: []
  },

  // 用户状态取消订阅函数
  _unsubscribeUserStore: null,

  onLoad() {

    this._unsubscribeUserStore = userStore.subscribe((state) => {
      this.setData({
        userInfo: state.userInfo,

      })

    this.loadBookingList()

    // 取消订阅，防止内存泄漏

      this._unsubscribeUserStore()
    }

  async loadBookingList() {

    const userStore = UserStore.getInstance()
    const userId = userStore.userId

      console.log('用户未登录')
      return
    }

    try {
      const bookingService = BookingService.getInstance()
      const list = await bookingService.getUserBookings(userId)
      this.setData({ bookingList: list })
    } catch (error) {
      console.error('加载预约列表失败:', error)
    }
  },

  async submitBooking() {

      return
    }

    // 获取用户ID（只获取ID，不传递完整用户信息）
    const userId = userStore.userId
    
    // 提交预约
    const bookingService = BookingService.getInstance()
    await bookingService.createBooking({
      userId: userId,  // 只传递ID
      // ... 其他预约信息
    })
  }
})
```

