## 一、架构概述

### 1.1 设计目标

实现一套灵活、可扩展的换肤系统，支持：

1. **动态加载** - 运行时加载不同主题资源包
2. **无缝切换** - 切换主题无需重启小程序
3. **资源隔离** - 各主题资源独立，互不干扰
4. **版本管理** - 支持主题版本控制和更新
5. **性能优化** - 缓存机制，减少网络请求

### 1.2 架构原则

## 三、核心模块设计

### 3.1 主题管理器 (ThemeManager)

```javascript
// utils/theme/ThemeManager.js

class ThemeManager {
  constructor() {
    this.currentTheme = null;
    this.themeCache = new Map();
    this.listeners = new Set();
    this.storageKey = 'theme_config';
    this.cachePrefix = 'theme_res_';
  }

  // 单例模式
  static getInstance() {
    if (!ThemeManager.instance) {
      ThemeManager.instance = new ThemeManager();
    }
    return ThemeManager.instance;
  }

  /**
   * 初始化主题系统
   * @param {Object} options - 配置选项
   */
  async init(options = {}) {
    this.options = {
      defaultTheme: 'modern-blue',
      cdnBase: 'https://cdn.example.com/themes/',
      localThemes: [], // 内置主题列表
      cacheEnabled: true,
      cacheExpiry: 7 * 24 * 60 * 60 * 1000, // 7天
      ...options
    };

    // 加载上次使用的主题
    const savedTheme = await this.getSavedTheme();
    await this.loadTheme(savedTheme || this.options.defaultTheme);
  }

  /**
   * 加载主题
   * @param {string} themeId - 主题ID
   * @param {Object} options - 加载选项
   */
  async loadTheme(themeId, options = {}) {
    try {
      // 检查缓存
      if (this.themeCache.has(themeId) && !options.force) {
        this.applyTheme(this.themeCache.get(themeId));
        return;
      }

      // 获取主题配置
      const themeConfig = await this.fetchThemeConfig(themeId);
      
      // 下载主题资源
      const resources = await this.downloadResources(themeConfig, options);
      
      // 构建主题对象
      const theme = {
        id: themeId,
        config: themeConfig,
        resources: resources,
        loadedAt: Date.now()
      };

      // 缓存主题
      this.themeCache.set(themeId, theme);
      
      // 应用主题
      this.applyTheme(theme);
      
      // 保存主题选择
      await this.saveTheme(themeId);

      // 通知监听器
      this.notifyListeners('themeChanged', theme);

    } catch (error) {
      console.error('加载主题失败:', error);
      this.notifyListeners('themeError', { themeId, error });
      throw error;
    }
  }

  /**
   * 获取主题配置
   * @param {string} themeId - 主题ID
   */
  async fetchThemeConfig(themeId) {
    // 检查本地缓存
    if (this.options.cacheEnabled) {
      const cached = await this.getCachedConfig(themeId);
      if (cached) return cached;
    }

    // 从网络获取
    const url = `${this.options.cdnBase}${themeId}/theme.config.json`;
    const response = await wx.request({ url, method: 'GET' });
    
    if (response.statusCode !== 200) {
      throw new Error(`获取主题配置失败: ${response.statusCode}`);
    }

    const config = response.data;
    
    // 缓存配置
    if (this.options.cacheEnabled) {
      await this.cacheConfig(themeId, config);
    }

    return config;
  }

  /**
   * 下载主题资源
   * @param {Object} config - 主题配置
   * @param {Object} options - 下载选项
   */
  async downloadResources(config, options = {}) {
    const resources = {};
    const { assets } = config;

    // 下载图片资源
    if (assets.images) {
      for (const [key, path] of Object.entries(assets.images)) {
        resources[key] = await this.downloadFile(path, options);
      }
    }

    // 下载图标资源
    if (assets.icons) {
      for (const [key, path] of Object.entries(assets.icons)) {
        resources[`icon_${key}`] = await this.downloadFile(path, options);
      }
    }

    return resources;
  }

  /**
   * 下载单个文件
   * @param {string} path - 文件路径
   * @param {Object} options - 下载选项
   */
  async downloadFile(path, options = {}) {
    const cacheKey = `${this.cachePrefix}${this.hashPath(path)}`;

    // 检查本地缓存
    if (this.options.cacheEnabled && !options.noCache) {
      const cached = await this.getCachedFile(cacheKey);
      if (cached) return cached;
    }

    // 下载文件
    const url = `${this.options.cdnBase}${path}`;
    const result = await wx.downloadFile({ url });

    if (result.statusCode !== 200) {
      throw new Error(`下载文件失败: ${url}`);
    }

    // 保存到本地缓存
    if (this.options.cacheEnabled) {
      await this.cacheFile(cacheKey, result.tempFilePath);
    }

    return result.tempFilePath;
  }

  /**
   * 应用主题
   * @param {Object} theme - 主题对象
   */
  applyTheme(theme) {
    this.currentTheme = theme;

    // 生成 CSS 变量
    const cssVars = this.generateCSSVariables(theme.config);
    
    // 注入全局样式
    this.injectGlobalStyles(cssVars);

    // 更新页面数据（触发重新渲染）
    this.updatePageData(theme);
  }

  /**
   * 生成 CSS 变量
   * @param {Object} config - 主题配置
   */
  generateCSSVariables(config) {
    const vars = [];

    // 颜色变量
    if (config.colors) {
      // 主色
      if (config.colors.primary) {
        for (const [key, value] of Object.entries(config.colors.primary)) {
          vars.push(`--primary-${key}: ${value};`);
        }
      }
      // 中性色
      if (config.colors.neutral) {
        for (const [key, value] of Object.entries(config.colors.neutral)) {
          vars.push(`--neutral-${key}: ${value};`);
        }
      }
      // 功能色
      if (config.colors.functional) {
        for (const [key, value] of Object.entries(config.colors.functional)) {
          vars.push(`--${key}: ${value};`);
        }
      }
    }

    // 间距变量
    if (config.spacing) {
      config.spacing.scale.forEach((value, index) => {
        vars.push(`--space-${index}: ${value}px;`);
      });
    }

    // 圆角变量
    if (config.radius) {
      for (const [key, value] of Object.entries(config.radius)) {
        vars.push(`--radius-${key}: ${value};`);
      }
    }

    // 阴影变量
    if (config.shadows) {
      for (const [key, value] of Object.entries(config.shadows)) {
        vars.push(`--shadow-${key}: ${value};`);
      }
    }

    return vars.join('\n');
  }

  /**
   * 注入全局样式
   * @param {string} cssVars - CSS变量字符串
   */
  injectGlobalStyles(cssVars) {
    // 在小程序中，通过更新全局数据来触发样式更新
    const app = getApp();
    if (app && app.globalData) {
      app.globalData.themeVars = cssVars;
      app.globalData.currentTheme = this.currentTheme;
    }
  }

  /**
   * 更新页面数据
   * @param {Object} theme - 主题对象
   */
  updatePageData(theme) {
    // 获取当前页面栈
    const pages = getCurrentPages();
    
    pages.forEach(page => {
      if (page && page.setData) {
        page.setData({
          theme: theme.config,
          themeResources: theme.resources
        });
      }
    });
  }

  /**
   * 获取当前主题
   */
  getCurrentTheme() {
    return this.currentTheme;
  }

  /**
   * 获取主题配置值
   * @param {string} path - 配置路径，如 'colors.primary.500'
   * @param {*} defaultValue - 默认值
   */
  getThemeValue(path, defaultValue = null) {
    if (!this.currentTheme) return defaultValue;
    
    const keys = path.split('.');
    let value = this.currentTheme.config;
    
    for (const key of keys) {
      if (value && typeof value === 'object' && key in value) {
        value = value[key];
      } else {
        return defaultValue;
      }
    }
    
    return value;
  }

  /**
   * 获取主题图片资源
   * @param {string} key - 资源key
   * @param {string} defaultPath - 默认路径
   */
  getThemeImage(key, defaultPath = '') {
    if (!this.currentTheme || !this.currentTheme.resources) {
      return defaultPath;
    }
    return this.currentTheme.resources[key] || defaultPath;
  }

  /**
   * 订阅主题变化
   * @param {Function} callback - 回调函数
   */
  subscribe(callback) {
    this.listeners.add(callback);
    return () => this.listeners.delete(callback);
  }

  /**
   * 通知监听器
   * @param {string} event - 事件类型
   * @param {*} data - 事件数据
   */
  notifyListeners(event, data) {
    this.listeners.forEach(callback => {
      try {
        callback(event, data);
      } catch (error) {
        console.error('主题监听器错误:', error);
      }
    });
  }

  /**
   * 保存主题选择
   * @param {string} themeId - 主题ID
   */
  async saveTheme(themeId) {
    return wx.setStorage({
      key: this.storageKey,
      data: { themeId, updatedAt: Date.now() }
    });
  }

  /**
   * 获取保存的主题
   */
  async getSavedTheme() {
    try {
      const result = await wx.getStorage({ key: this.storageKey });
      return result.data?.themeId;
    } catch {
      return null;
    }
  }

  /**
   * 缓存配置
   */
  async cacheConfig(themeId, config) {
    const key = `${this.cachePrefix}config_${themeId}`;
    await wx.setStorage({
      key,
      data: { config, cachedAt: Date.now() }
    });
  }

  /**
   * 获取缓存的配置
   */
  async getCachedConfig(themeId) {
    try {
      const key = `${this.cachePrefix}config_${themeId}`;
      const result = await wx.getStorage({ key });
      const { config, cachedAt } = result.data;
      
      // 检查是否过期
      if (Date.now() - cachedAt > this.options.cacheExpiry) {
        return null;
      }
      
      return config;
    } catch {
      return null;
    }
  }

  /**
   * 缓存文件
   */
  async cacheFile(key, filePath) {
    // 保存文件到本地存储
    const fs = wx.getFileSystemManager();
    const localPath = `${wx.env.USER_DATA_PATH}/${key}`;
    
    try {
      fs.copyFileSync(filePath, localPath);
      await wx.setStorage({
        key: `${key}_meta`,
        data: { path: localPath, cachedAt: Date.now() }
      });
    } catch (error) {
      console.error('缓存文件失败:', error);
    }
  }

  /**
   * 获取缓存的文件
   */
  async getCachedFile(key) {
    try {
      const meta = await wx.getStorage({ key: `${key}_meta` });
      const { path, cachedAt } = meta.data;
      
      // 检查是否过期
      if (Date.now() - cachedAt > this.options.cacheExpiry) {
        return null;
      }
      
      // 检查文件是否存在
      const fs = wx.getFileSystemManager();
      fs.accessSync(path);
      
      return path;
    } catch {
      return null;
    }
  }

  /**
   * 清理缓存
   */
  async clearCache() {
    // 清理主题缓存
    this.themeCache.clear();
    
    // 清理本地存储
    const keys = await wx.getStorageInfo();
    const themeKeys = keys.keys.filter(k => k.startsWith(this.cachePrefix));
    
    for (const key of themeKeys) {
      await wx.removeStorage({ key });
    }
  }

  /**
   * 路径哈希
   */
  hashPath(path) {
    // 简单的哈希函数
    let hash = 0;
    for (let i = 0; i < path.length; i++) {
      const char = path.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return Math.abs(hash).toString(16);
  }
}
```

### 3.2 主题Hook/Composable

```javascript
// utils/theme/useTheme.js

import { ref, computed, onMounted, onUnmounted } from 'vue';
import themeManager from './ThemeManager';

/**
 * 主题组合式函数
 * 在 Vue3 组合式 API 中使用
 */
export function useTheme() {
  const currentTheme = ref(null);
  const isLoading = ref(false);
  const error = ref(null);

  // 计算属性
  const themeColors = computed(() => currentTheme.value?.config?.colors || {});
  const themeAssets = computed(() => currentTheme.value?.resources || {});

  // 订阅主题变化
  let unsubscribe = null;

  onMounted(() => {
    // 初始化当前主题
    currentTheme.value = themeManager.getCurrentTheme();

    // 订阅主题变化
    unsubscribe = themeManager.subscribe((event, data) => {
      if (event === 'themeChanged') {
        currentTheme.value = data;
        error.value = null;
      } else if (event === 'themeError') {
        error.value = data.error;
      }
    });
  });

  onUnmounted(() => {
    if (unsubscribe) {
      unsubscribe();
    }
  });

  // 方法
  const loadTheme = async (themeId, options = {}) => {
    isLoading.value = true;
    error.value = null;
    
    try {
      await themeManager.loadTheme(themeId, options);
    } catch (err) {
      error.value = err;
    } finally {
      isLoading.value = false;
    }
  };

  const getColor = (path, defaultValue = '') => {
    return themeManager.getThemeValue(`colors.${path}`, defaultValue);
  };

  const getAsset = (key, defaultPath = '') => {
    return themeManager.getThemeImage(key, defaultPath);
  };

  return {
    currentTheme,
    isLoading,
    error,
    themeColors,
    themeAssets,
    loadTheme,
    getColor,
    getAsset
  };
}

export function useThemeInPage(page) {
  const themeData = {
    theme: null,
    themeResources: null
  };

  // 页面加载时
  page.onLoad = function(options) {
    // 初始化主题数据
    const currentTheme = themeManager.getCurrentTheme();
    if (currentTheme) {
      this.setData({
        theme: currentTheme.config,
        themeResources: currentTheme.resources
      });
    }

    // 订阅主题变化
    this._themeUnsubscribe = themeManager.subscribe((event, data) => {
      if (event === 'themeChanged') {
        this.setData({
          theme: data.config,
          themeResources: data.resources
        });
      }
    });

    // 调用原 onLoad
    if (page._originalOnLoad) {
      page._originalOnLoad.call(this, options);
    }
  };

  // 页面卸载时
  page.onUnload = function() {
    // 取消订阅
    if (this._themeUnsubscribe) {
      this._themeUnsubscribe();
    }

    // 调用原 onUnload
    if (page._originalOnUnload) {
      page._originalOnUnload.call(this);
    }
  };

  // 保存原始生命周期
  page._originalOnLoad = page.onLoad;
  page._originalOnUnload = page.onUnload;

  return page;
}
```

```javascript
// utils/theme/ThemeVersion.js

class ThemeVersion {
  /**
   * 检查主题更新
   * @param {string} themeId - 主题ID
   * @param {string} currentVersion - 当前版本
   */
  static async checkUpdate(themeId, currentVersion) {
    try {
      const url = `https://cdn.example.com/themes/${themeId}/version.json`;
      const response = await wx.request({ url, method: 'GET' });
      
      if (response.statusCode === 200) {
        const { version, changelog, forceUpdate } = response.data;
        
        return {
          hasUpdate: this.compareVersion(version, currentVersion) > 0,
          version,
          changelog,
          forceUpdate
        };
      }
      
      return { hasUpdate: false };
    } catch (error) {
      console.error('检查主题更新失败:', error);
      return { hasUpdate: false, error };
    }
  }

  /**
   * 版本号比较
   * @param {string} v1 - 版本1
   * @param {string} v2 - 版本2
   * @returns {number} 1: v1>v2, 0: v1=v2, -1: v1<v2
   */
  static compareVersion(v1, v2) {
    const parts1 = v1.split('.').map(Number);
    const parts2 = v2.split('.').map(Number);
    
    for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
      const p1 = parts1[i] || 0;
      const p2 = parts2[i] || 0;
      
      if (p1 > p2) return 1;
      if (p1 < p2) return -1;
    }
    
    return 0;
  }
}

export default ThemeVersion;
```

### 4.3 资源包验证

```javascript
// utils/theme/ThemeValidator.js

class ThemeValidator {
  static REQUIRED_FIELDS = [
    'theme.id',
    'theme.name',
    'theme.version',
    'colors.primary',
    'colors.neutral'
  ];

  /**
   * 验证主题配置
   * @param {Object} config - 主题配置
   */
  static validate(config) {
    const errors = [];

    // 检查必填字段
    for (const field of this.REQUIRED_FIELDS) {
      if (!this.getNestedValue(config, field)) {
        errors.push(`缺少必填字段: ${field}`);
      }
    }

    // 验证颜色格式
    if (config.colors) {
      this.validateColors(config.colors, errors);
    }

    // 验证版本号格式
    if (config.theme?.version) {
      if (!this.isValidVersion(config.theme.version)) {
        errors.push('版本号格式不正确，应为 x.y.z');
      }
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }

  /**
   * 验证颜色配置
   */
  static validateColors(colors, errors) {
    const colorRegex = /^#([A-Fa-f0-9]{6}|[A-Fa-f0-9]{3})$/;

    const validateColorObject = (obj, prefix) => {
      for (const [key, value] of Object.entries(obj)) {
        if (typeof value === 'string' && !colorRegex.test(value)) {
          errors.push(`${prefix}.${key} 颜色格式不正确: ${value}`);
        } else if (typeof value === 'object') {
          validateColorObject(value, `${prefix}.${key}`);
        }
      }
    };

    validateColorObject(colors, 'colors');
  }

  /**
   * 验证版本号格式
   */
  static isValidVersion(version) {
    return /^\d+\.\d+\.\d+/.test(version);
  }

  /**
   * 获取嵌套值
   */
  static getNestedValue(obj, path) {
    return path.split('.').reduce((current, key) => {
      return current && current[key] !== undefined ? current[key] : undefined;
    }, obj);
  }
}

export default ThemeValidator;
```

### 6.1 缓存策略

```javascript
// utils/theme/ThemeCache.js

class ThemeCache {
  constructor(options = {}) {
    this.maxSize = options.maxSize || 50 * 1024 * 1024; // 50MB
    this.maxAge = options.maxAge || 7 * 24 * 60 * 60 * 1000; // 7天
  }

  /**
   * 计算缓存大小
   */
  async calculateSize() {
    const info = await wx.getStorageInfo();
    return info.currentSize;
  }

  /**
   * 清理过期缓存
   */
  async cleanup() {
    const keys = await wx.getStorageInfo();
    const now = Date.now();

    for (const key of keys.keys) {
      if (key.startsWith('theme_res_')) {
        try {
          const meta = await wx.getStorage({ key: `${key}_meta` });
          if (now - meta.data.cachedAt > this.maxAge) {
            await this.remove(key);
          }
        } catch {
          // 忽略错误
        }
      }
    }
  }

  /**
   * 移除缓存
   */
  async remove(key) {
    try {
      // 删除文件
      const meta = await wx.getStorage({ key: `${key}_meta` });
      const fs = wx.getFileSystemManager();
      fs.unlinkSync(meta.data.path);
    } catch {
      // 文件可能不存在
    }

    // 删除存储记录
    await wx.removeStorage({ key });
    await wx.removeStorage({ key: `${key}_meta` });
  }
}

export default ThemeCache;
```

```javascript
// utils/theme/ThemePreloader.js

class ThemePreloader {
  /**
   * 预加载主题资源
   * @param {string} themeId - 主题ID
   * @param {Array} priorityResources - 优先加载的资源
   */
  static async preload(themeId, priorityResources = []) {
    const themeManager = require('./ThemeManager').default;

    try {
      // 获取主题配置
      const config = await themeManager.fetchThemeConfig(themeId);

      // 优先加载关键资源
      for (const key of priorityResources) {
        const path = config.assets?.images?.[key] || config.assets?.icons?.[key];
        if (path) {
          await themeManager.downloadFile(path, { noCache: false });
        }
      }

      return { success: true };
    } catch (error) {
      console.error('预加载失败:', error);
      return { success: false, error };
    }
  }

  /**
   * 智能预加载
   * 根据用户行为预测可能切换的主题
   */
  static async smartPreload() {
    // 获取用户偏好
    const preference = await this.getUserPreference();
    
    // 预加载用户可能选择的主题
    const likelyThemes = this.predictThemes(preference);
    
    for (const themeId of likelyThemes) {
      // 低优先级预加载，不阻塞主线程
      this.preload(themeId, ['logo', 'bgHome']).catch(() => {});
    }
  }

  /**
   * 获取用户偏好
   */
  static async getUserPreference() {
    try {
      const result = await wx.getStorage({ key: 'theme_preference' });
      return result.data;
    } catch {
      return null;
    }
  }

  /**
   * 预测用户可能选择的主题
   */
  static predictThemes(preference) {
    // 基于时间、场景等预测
    const hour = new Date().getHours();
    const themes = [];

    if (hour >= 18 || hour <= 6) {
      themes.push('dark-pro'); // 夜间推荐深色主题
    }

    if (preference?.favorite) {
      themes.push(preference.favorite);
    }

    return themes;
  }
}

export default ThemePreloader;
```

### 9.1 开发建议

1. **渐进式增强** - 先实现基础主题，再逐步添加高级功能

3. **按需加载** - 只加载当前页面需要的资源
4. **版本兼容** - 主题配置添加版本号，支持向后兼容

### 9.2 性能建议

2. **懒加载** - 非首屏资源延迟加载
3. **缓存优先** - 优先使用本地缓存，减少网络请求
4. **增量更新** - 主题更新时只下载变更的资源
