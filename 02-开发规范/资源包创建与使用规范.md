### 1.3 资源文件规范

#### 图片资源规范

| 资源名称 | 尺寸要求 | 格式 | 大小限制 | 用途 |
|---------|---------|------|---------|------|
| logo.png | 200×200 | PNG | < 50KB | 应用Logo |
| bg-home.png | 750×400 | PNG/JPG | < 200KB | 首页背景 |
| bg-profile.png | 750×300 | PNG/JPG | < 150KB | 个人中心背景 |
| empty-*.png | 200×200 | PNG | < 30KB | 空状态插图 |
| error-*.png | 200×200 | PNG | < 30KB | 错误状态插图 |
| tab-*.png | 48×48 | PNG | < 5KB | Tab栏图标 |
| icon-*.png | 48×48 | PNG | < 5KB | 功能图标 |

#### 图标设计规范

1. **风格统一** - 所有图标使用相同样式（线性/填充）
2. **视觉平衡** - 保持图标视觉重心一致
3. **尺寸规范** - 使用 48×48 画布，图标内容在 44×44 安全区域内
4. **格式要求** - PNG 格式，支持透明背景
5. **命名规范** - 使用 kebab-case，如 `icon-arrow-right.png`

### 1.4 打包规范

#### 打包脚本示例

```javascript
// scripts/build-theme.js

const fs = require('fs');
const path = require('path');
const crypto = require('crypto');
const JSZip = require('jszip');

class ThemeBuilder {
  constructor(sourceDir, outputDir) {
    this.sourceDir = sourceDir;
    this.outputDir = outputDir;
  }

  async build() {
    console.log('开始构建主题包...');

    // 1. 验证配置
    await this.validateConfig();

    // 2. 生成资源清单
    await this.generateManifest();

    // 3. 优化资源
    await this.optimizeAssets();

    // 4. 打包
    await this.createPackage();

    console.log('主题包构建完成！');
  }

  async validateConfig() {
    const configPath = path.join(this.sourceDir, 'theme.config.json');
    const config = JSON.parse(fs.readFileSync(configPath, 'utf8'));

    // 验证必填字段
    const required = ['theme.id', 'theme.name', 'theme.version'];
    for (const field of required) {
      const value = field.split('.').reduce((obj, key) => obj?.[key], config);
      if (!value) {
        throw new Error(`缺少必填字段: ${field}`);
      }
    }

    // 验证版本号格式
    if (!/^\d+\.\d+\.\d+/.test(config.theme.version)) {
      throw new Error('版本号格式不正确');
    }

    console.log('✓ 配置验证通过');
  }

  async generateManifest() {
    const manifest = {
      version: '1.0.0',
      generatedAt: new Date().toISOString(),
      resources: []
    };

    const assetsDir = path.join(this.sourceDir, 'assets');
    
    const scanDir = (dir, basePath = '') => {
      const items = fs.readdirSync(dir);
      
      for (const item of items) {
        const fullPath = path.join(dir, item);
        const relativePath = path.join(basePath, item);
        
        if (fs.statSync(fullPath).isDirectory()) {
          scanDir(fullPath, relativePath);
        } else {
          const content = fs.readFileSync(fullPath);
          const hash = crypto.createHash('sha256').update(content).digest('hex');
          
          manifest.resources.push({
            path: relativePath,
            type: this.getResourceType(item),
            size: content.length,
            hash: `sha256:${hash.slice(0, 16)}`,
            url: `assets/${relativePath}`
          });
        }
      }
    };

    if (fs.existsSync(assetsDir)) {
      scanDir(assetsDir);
    }

    manifest.totalSize = manifest.resources.reduce((sum, r) => sum + r.size, 0);

    fs.writeFileSync(
      path.join(this.sourceDir, 'manifest.json'),
      JSON.stringify(manifest, null, 2)
    );

    console.log('✓ 资源清单生成完成');
  }

  getResourceType(filename) {
    const ext = path.extname(filename).toLowerCase();
    const types = {
      '.png': 'image',
      '.jpg': 'image',
      '.jpeg': 'image',
      '.gif': 'image',
      '.webp': 'image',
      '.svg': 'image',
      '.woff': 'font',
      '.woff2': 'font',
      '.ttf': 'font'
    };
    return types[ext] || 'unknown';
  }

  async optimizeAssets() {
    // 图片压缩
    // 可以使用 sharp 等库进行压缩
    console.log('✓ 资源优化完成');
  }

  async createPackage() {
    const config = JSON.parse(
      fs.readFileSync(path.join(this.sourceDir, 'theme.config.json'), 'utf8')
    );

    const zip = new JSZip();

    // 添加配置文件
    zip.file('theme.config.json', fs.readFileSync(path.join(this.sourceDir, 'theme.config.json')));
    zip.file('manifest.json', fs.readFileSync(path.join(this.sourceDir, 'manifest.json')));

    // 添加资源文件
    const addFilesToZip = (dir, zipFolder) => {
      const items = fs.readdirSync(dir);
      
      for (const item of items) {
        const fullPath = path.join(dir, item);
        
        if (fs.statSync(fullPath).isDirectory()) {
          const newFolder = zipFolder.folder(item);
          addFilesToZip(fullPath, newFolder);
        } else {
          zipFolder.file(item, fs.readFileSync(fullPath));
        }
      }
    };

    const assetsDir = path.join(this.sourceDir, 'assets');
    if (fs.existsSync(assetsDir)) {
      const assetsFolder = zip.folder('assets');
      addFilesToZip(assetsDir, assetsFolder);
    }

    // 生成 ZIP
    const content = await zip.generateAsync({ type: 'nodebuffer' });
    const outputName = `${config.theme.id}-v${config.theme.version}.zip`;
    
    if (!fs.existsSync(this.outputDir)) {
      fs.mkdirSync(this.outputDir, { recursive: true });
    }

    fs.writeFileSync(path.join(this.outputDir, outputName), content);

    console.log(`✓ 主题包已生成: ${outputName}`);
  }
}

// 使用

builder.build().catch(console.error);
```

### 2.1 小程序端使用

#### 初始化主题系统

```javascript
// app.js

import themeManager from './utils/theme/ThemeManager';

App({
  globalData: {
    theme: null
  },

  async onLaunch() {
    // 初始化主题管理器
    await themeManager.init({
      // CDN 基础路径
      cdnBase: 'https://your-cdn.com/themes/',
      
      // 默认主题
      defaultTheme: 'modern-blue',
      
      // 内置主题（离线可用）
      localThemes: ['modern-blue'],
      
      // 缓存配置
      cacheEnabled: true,
      cacheExpiry: 7 * 24 * 60 * 60 * 1000, // 7天
      
      // 预加载配置
      preload: {
        enabled: true,
        resources: ['logo', 'bgHome']
      }
    });

    // 保存到全局
    this.globalData.theme = themeManager.getCurrentTheme();
  }
});
```

```javascript
  // 切换主题
  async switchTheme(e) {
    const themeId = e.currentTarget.dataset.theme;

      this.setData({
        theme: currentTheme?.config,
        themeResources: currentTheme?.resources
      });
      
      wx.showToast({ title: '切换成功', icon: 'success' });
    } catch (error) {
      wx.showToast({ title: '切换失败', icon: 'error' });
    } finally {
      wx.hideLoading();
    }
  },

  // 获取主题颜色
  getThemeColor(path, defaultValue = '') {
    const themeManager = require('../../utils/theme/ThemeManager').default;
    return themeManager.getThemeValue(`colors.${path}`, defaultValue);
  },

  // 获取主题图片
  getThemeImage(key, defaultPath = '') {
    const themeManager = require('../../utils/theme/ThemeManager').default;
    return themeManager.getThemeImage(key, defaultPath);
  }
```

#### 平滑过渡动画

```javascript
// utils/theme/ThemeTransition.js

class ThemeTransition {
  /**
   * 带过渡效果的主题切换
   */
  static async switchWithTransition(themeId, options = {}) {
    const pages = getCurrentPages();
    const currentPage = pages[pages.length - 1];

    // 1. 显示遮罩
    currentPage.setData({
      themeTransition: {
        show: true,
        opacity: 0
      }
    });

    // 2. 淡入遮罩
    await this.animate(currentPage, {
      'themeTransition.opacity': 1
    }, 200);

    // 3. 切换主题
    const themeManager = require('./ThemeManager').default;
    await themeManager.loadTheme(themeId);

    // 4. 淡出遮罩
    await this.animate(currentPage, {
      'themeTransition.opacity': 0
    }, 200);

    // 5. 隐藏遮罩
    currentPage.setData({
      'themeTransition.show': false
    });
  }

  static animate(page, properties, duration) {
    return new Promise(resolve => {
      const animation = wx.createAnimation({
        duration,
        timingFunction: 'ease'
      });

      // 应用动画
      for (const [key, value] of Object.entries(properties)) {
        // 解析路径
        const keys = key.split('.');
        let target = animation;
        
        for (let i = 0; i < keys.length - 1; i++) {
          target = target[keys[i]];
        }
        
        target[keys[keys.length - 1]](value);
      }

      page.setData({
        animationData: animation.export()
      });

      setTimeout(resolve, duration);
    });
  }
}

export default ThemeTransition;
```

### 2.3 错误处理

```javascript
// utils/theme/ThemeErrorHandler.js

class ThemeErrorHandler {
  static handle(error, context = {}) {
    console.error('主题错误:', error);

    switch (error.code) {
      case 'NETWORK_ERROR':
        this.handleNetworkError(error, context);
        break;
      case 'INVALID_CONFIG':
        this.handleInvalidConfig(error, context);
        break;
      case 'RESOURCE_NOT_FOUND':
        this.handleResourceNotFound(error, context);
        break;
      case 'VERSION_INCOMPATIBLE':
        this.handleVersionIncompatible(error, context);
        break;
      default:
        this.handleUnknownError(error, context);
    }
  }

  static handleNetworkError(error, context) {
    wx.showModal({
      title: '网络错误',
      content: '主题加载失败，是否使用默认主题？',
      success: (res) => {
        if (res.confirm) {
          this.fallbackToDefault(context);
        }
      }
    });
  }

  static handleInvalidConfig(error, context) {
    wx.showToast({
      title: '主题配置错误',
      icon: 'error'
    });
    this.fallbackToDefault(context);
  }

  static handleResourceNotFound(error, context) {
    console.warn('资源未找到:', error.resource);
    // 使用占位图
  }

  static handleVersionIncompatible(error, context) {
    wx.showModal({
      title: '版本不兼容',
      content: '该主题需要更新版本的小程序，是否更新？',
      success: (res) => {
        if (res.confirm) {
          // 引导用户更新
        }
      }
    });
  }

  static handleUnknownError(error, context) {
    wx.showToast({
      title: '主题加载失败',
      icon: 'error'
    });
    this.fallbackToDefault(context);
  }

  static fallbackToDefault(context) {
    const themeManager = require('./ThemeManager').default;
    themeManager.loadTheme('modern-blue', { force: true });
  }
}

export default ThemeErrorHandler;
```

### 5.2 常见问题

**Q: 主题切换后页面没有更新？**  
A: 确保页面订阅了主题变化事件，或手动调用 setData 更新。

**Q: 图片资源加载失败？**  
A: 检查图片路径是否正确，网络是否可用，是否有跨域问题。

**Q: 如何支持暗黑模式？**  
A: 在主题配置中定义深色模式的颜色变量，根据系统设置切换。

**Q: 主题包大小限制？**  
A: 建议单个主题包不超过 2MB，图片资源需要压缩。
